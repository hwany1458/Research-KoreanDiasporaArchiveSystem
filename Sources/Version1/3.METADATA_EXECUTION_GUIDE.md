## 3. 관계 추론 엔진 (relation_engine.py)

### 파일: `src/core/metadata/relation_engine.py`

### 3.1 주요 클래스

| 클래스 | 기능 | 기술 |
|--------|------|------|
| `FaceRelationEngine` | 얼굴 기반 인물 연결 | face_recognition |
| `TemporalRelationEngine` | 시간 기반 관계 | 정규식 |
| `SpatialRelationEngine` | 장소 기반 관계 | 지명 정규화 |
| `TopicRelationEngine` | 주제 기반 관계 | sentence-transformers |
| `EventRelationEngine` | 행사 기반 관계 | 키워드 매칭 |
| `RelationInferenceEngine` | **통합 엔진** | 전체 통합 |

### 3.2 관계 유형

```python
DEPICTS = "depicts"              # 인물 등장
SAME_PERIOD = "same_period"      # 같은 시기
SAME_LOCATION = "same_location"  # 같은 장소
SAME_EVENT = "same_event"        # 같은 행사
SAME_TOPIC = "same_topic"        # 같은 주제
SIMILAR_CONTENT = "similar_content"  # 유사 내용
```

### 3.3 Python 사용법

```python
from src.core.metadata.relation_engine import (
    RelationInferenceEngine, RelationType,
    TemporalRelationEngine, SpatialRelationEngine, TopicRelationEngine
)
from src.core.metadata.schema import MetadataFactory

# ========== 1. 통합 엔진 사용 (권장) ==========

engine = RelationInferenceEngine({"device": "cpu"})
engine.load_models()

# 테스트 데이터
items = []

item1 = MetadataFactory.create_for_image("1975년 가족사진", "p1.jpg", 1200, 800)
item1.dublin_core.date = "1975"
item1.dublin_core.coverage_spatial = "Los Angeles"
item1.tags = ["가족", "LA"]
items.append(item1)

item2 = MetadataFactory.create_for_image("1976년 가족사진", "p2.jpg", 1200, 800)
item2.dublin_core.date = "1976"
item2.dublin_core.coverage_spatial = "Los Angeles"
item2.tags = ["가족", "LA"]
items.append(item2)

item3 = MetadataFactory.create_for_image("1975년 결혼식", "p3.jpg", 1200, 800)
item3.dublin_core.date = "1975"
item3.dublin_core.coverage_spatial = "Seoul"
item3.tags = ["결혼식"]
items.append(item3)

# 관계 추론
relations = engine.infer_relations(items)

print(f"발견된 관계: {len(relations)}개")
for rel in relations:
    print(f"  [{rel.relation_type.value}] 신뢰도:{rel.confidence:.2f}")
    print(f"    {rel.evidence}")

# 특정 유형만
temporal_only = engine.infer_relations(items, relation_types=["temporal"])


# ========== 2. 특정 자료의 관계 조회 ==========

item_id = items[0].item_id
item_relations = engine.get_relations_for_item(item_id)
print(f"\n자료의 관계: {len(item_relations)}개")

# 관련 자료 (신뢰도 순)
related = engine.get_related_items(item_id, min_confidence=0.5)
for other_id, rel in related:
    print(f"  {other_id[:8]}... ({rel.relation_type.value})")


# ========== 3. 개별 엔진 사용 ==========

# 시간 기반
temporal = TemporalRelationEngine()
year, month, day = temporal.parse_date("1975년 6월 15일")
print(f"파싱: {year}년 {month}월 {day}일")

temporal_rels = temporal.find_temporal_relations(items, year_tolerance=2)
print(f"시간 관계: {len(temporal_rels)}개")

# 장소 기반
spatial = SpatialRelationEngine()
print(f"정규화: Los Angeles -> {spatial.normalize_location('Los Angeles')}")
print(f"정규화: 로스앤젤레스 -> {spatial.normalize_location('로스앤젤레스')}")

spatial_rels = spatial.find_spatial_relations(items)
print(f"장소 관계: {len(spatial_rels)}개")

# 주제 기반
topic = TopicRelationEngine("cpu")
topic.load_models()

similarity = topic.compute_similarity(
    "미국 이민 후 첫 가족 사진",
    "1970년대 LA에서 찍은 가족 사진"
)
print(f"텍스트 유사도: {similarity:.2f}")


# ========== 4. 내보내기/가져오기 ==========

engine.export_relations("data/relations.json")

new_engine = RelationInferenceEngine()
new_engine.import_relations("data/relations.json")
print(f"가져온 관계: {len(new_engine.relations)}개")
```

```python
ITEM = "Item"           # 자료
PERSON = "Person"       # 인물
LOCATION = "Location"   # 장소
EVENT = "Event"         # 행사
TOPIC = "Topic"         # 주제/태그
```

### 4.3 Python 사용법

```python
from src.core.metadata.knowledge_graph import KnowledgeGraph, NodeType
from src.core.metadata.schema import MetadataFactory, PersonInfo, LocationInfo, EventInfo
from src.core.metadata.relation_engine import Relation, RelationType

# ========== 1. 초기화 ==========

kg = KnowledgeGraph({
    "uri": "bolt://localhost:7687",
    "user": "neo4j",
    "password": "neo4j123"
})

connected = kg.connect()
if connected:
    print("Neo4j 연결됨")
    kg.create_indexes()
else:
    print("인메모리 모드")


# ========== 2. 메타데이터에서 그래프 자동 구축 ==========

metadata = MetadataFactory.create_for_image("가족 사진", "photo.jpg", 1200, 800)

metadata.diaspora.persons = [
    PersonInfo(name_korean="김철수", generation=1),
    PersonInfo(name_korean="김영희", generation=1)
]
metadata.diaspora.locations = [
    LocationInfo(name_korean="코리아타운", city="Los Angeles")
]
metadata.diaspora.events = [
    EventInfo(name="가족 모임", event_type="community_event")
]
metadata.tags = ["가족", "LA", "1970년대"]

# 한번에 구축
kg.build_from_metadata(metadata)
print("그래프 구축 완료")


# ========== 3. 개별 노드/엣지 생성 ==========

# 노드 생성
item_node = kg.create_item_node(metadata)
person_node = kg.create_person_node(metadata.diaspora.persons[0])
location_node = kg.create_location_node(metadata.diaspora.locations[0])
event_node = kg.create_event_node(metadata.diaspora.events[0])
topic_node = kg.create_topic_node("이민역사")

# 엣지 생성
kg.create_depicts_edge(metadata.item_id, metadata.diaspora.persons[0].person_id)
kg.create_located_at_edge(metadata.item_id, metadata.diaspora.locations[0].location_id)
kg.create_part_of_event_edge(metadata.item_id, metadata.diaspora.events[0].event_id)
kg.create_tagged_with_edge(metadata.item_id, topic_node.node_id)


# ========== 4. 쿼리 ==========

# 노드 조회
node = kg.get_node(metadata.item_id)
print(f"노드: {node.node_type} - {node.properties}")

# 이웃 조회
neighbors = kg.get_neighbors(metadata.item_id)
print(f"\n이웃 ({len(neighbors)}개):")
for neighbor, edge in neighbors:
    name = neighbor.properties.get('name_korean') or neighbor.properties.get('title')
    print(f"  [{edge.edge_type}] {neighbor.node_type.value}: {name}")

# 특정 관계만
person_neighbors = kg.get_neighbors(metadata.item_id, edge_types=["DEPICTS"])


# ========== 5. 경로 탐색 ==========

paths = kg.find_path(metadata.item_id, metadata.diaspora.persons[0].person_id, max_depth=3)
print(f"\n경로 ({len(paths)}개):")
for path in paths:
    print(f"  {' -> '.join([p[:8] for p in path])}")


# ========== 6. 관련 자료 추천 ==========

related = kg.get_related_items(metadata.item_id, max_depth=2, limit=10)
print(f"\n관련 자료 ({len(related)}개):")
for r in related:
    print(f"  {r['title']} (연결: {r['connection_count']})")


# ========== 7. 검색 ==========

# 인물로 검색
results = kg.search_by_person("김철수")
print(f"\n'김철수' 자료: {len(results)}개")

# 장소로 검색
results = kg.search_by_location("Los Angeles")
print(f"'Los Angeles' 자료: {len(results)}개")

# 행사로 검색
results = kg.search_by_event("wedding")
print(f"결혼식 자료: {len(results)}개")


# ========== 8. 통계 ==========

stats = kg.get_statistics()
print(f"\n통계: 노드={stats['nodes']}, 엣지={stats['edges']}")


# ========== 9. 내보내기/가져오기 ==========

kg.export_graph("data/graph.json")

new_kg = KnowledgeGraph()
new_kg.import_graph("data/graph.json")
print(f"가져온 노드: {len(new_kg.nodes)}개")

kg.disconnect()
```

## 5. 통합 저장소 (storage.py)

### 파일: `src/core/metadata/storage.py`

### 5.1 주요 클래스

| 클래스 | 기능 | 서비스 |
|--------|------|--------|
| `PostgreSQLStorage` | 메타데이터 저장 | PostgreSQL |
| `ElasticsearchStorage` | 전문 검색 | Elasticsearch |
| `MinIOStorage` | 파일 저장 | MinIO/S3 |
| `UnifiedStorage` | **통합 관리자** | 전체 |

### 5.2 Python 사용법

```python
from src.core.metadata.storage import UnifiedStorage, PostgreSQLStorage, ElasticsearchStorage, MinIOStorage
from src.core.metadata.schema import MetadataFactory

# ========== 1. 통합 저장소 (권장) ==========

config = {
    "postgresql": {
        "host": "localhost", "port": 5432,
        "database": "diaspora_archive",
        "user": "postgres", "password": "postgres123"
    },
    "elasticsearch": {"hosts": ["localhost:9200"]},
    "minio": {
        "endpoint": "localhost:9000",
        "access_key": "minioadmin",
        "secret_key": "minioadmin123",
        "bucket": "diaspora-archive"
    },
    "local_path": "./data",
    "use_local_fallback": True  # 서비스 없으면 로컬 사용
}

storage = UnifiedStorage(config)
status = storage.connect_all()
print(f"연결: {status}")

# 메타데이터 준비
metadata = MetadataFactory.create_for_image("가족 사진", "photo.jpg", 1200, 800)
metadata.dublin_core.description = "미국 이민 후 첫 사진"
metadata.tags = ["가족", "LA"]

# 저장
result = storage.save(metadata, file_path="photo.jpg")
print(f"저장: {result}")

# 조회
loaded = storage.get(metadata.item_id)
print(f"제목: {loaded.dublin_core.title}")

# 검색
results = storage.search("가족 사진")
print(f"검색: {results['total']}건")

# 파일 URL
url = storage.get_file_url(metadata.storage_path)
print(f"URL: {url}")

# 통계
stats = storage.get_statistics()
print(f"통계: {stats}")

storage.disconnect_all()


# ========== 2. PostgreSQL 개별 사용 ==========

pg = PostgreSQLStorage({
    "host": "localhost", "port": 5432,
    "database": "diaspora_archive",
    "user": "postgres", "password": "postgres123"
})

if pg.connect():
    pg.create_tables()
    pg.save_metadata(metadata)
    loaded = pg.get_metadata(metadata.item_id)
    results = pg.search("가족", material_type="photograph", limit=20)
    pg.disconnect()


# ========== 3. Elasticsearch 개별 사용 ==========

es = ElasticsearchStorage({"hosts": ["localhost:9200"]})

if es.connect():
    es.create_index()
    es.index_metadata(metadata)
    results = es.search("이민 가족", filters={"material_type": "photograph"})
    print(f"검색: {results['total']}건")
    
    suggestions = es.suggest("가족", field="title")
    print(f"자동완성: {suggestions}")


# ========== 4. MinIO 개별 사용 ==========

minio = MinIOStorage({
    "endpoint": "localhost:9000",
    "access_key": "minioadmin",
    "secret_key": "minioadmin123",
    "bucket": "diaspora-archive"
})

if minio.connect():
    path = minio.upload_file("photo.jpg", "images/photo.jpg")
    print(f"업로드: {path}")
    
    minio.download_file("images/photo.jpg", "download.jpg")
    
    url = minio.get_url("images/photo.jpg", expires_hours=24)
    print(f"URL: {url}")
    
    files = minio.list_files("images/")
    print(f"파일: {files}")


# ========== 5. 로컬 모드 (서비스 없이) ==========

storage = UnifiedStorage({
    "local_path": "./data",
    "use_local_fallback": True
})
storage.connect_all()  # 실패해도 OK

storage.save(metadata)  # 로컬 파일로 저장
# -> data/metadata/{item_id}.json
# -> data/files/photo.jpg
```

```python
"""
3단계 메타데이터 통합 파이프라인
"""

from src.core.metadata import (
    MetadataFactory, PersonInfo, LocationInfo, EventInfo,
    MaterialType, DiasporaGeneration, DiasporaRegion,
    RelationInferenceEngine, KnowledgeGraph, UnifiedStorage
)
import os


def run_pipeline(input_dir, output_dir):
    os.makedirs(output_dir, exist_ok=True)
    
    # ===== 1. 초기화 =====
    print("1. 초기화")
    
    config = {
        "local_path": output_dir,
        "use_local_fallback": True
    }
    
    storage = UnifiedStorage(config)
    storage.connect_all()
    
    kg = KnowledgeGraph()
    kg.connect()
    
    engine = RelationInferenceEngine({"device": "cpu"})
    engine.load_models()
    
    
    # ===== 2. 메타데이터 생성 =====
    print("\n2. 메타데이터 생성")
    
    items = []
    
    # 사진
    photo = MetadataFactory.create_for_image(
        title="1975년 LA 가족 사진",
        file_path=f"{input_dir}/photo.jpg",
        width=1200, height=800
    )
    photo.dublin_core.date = "1975"
    photo.dublin_core.coverage_spatial = "Los Angeles"
    photo.tags = ["가족", "LA", "1970년대"]
    photo.diaspora.persons = [
        PersonInfo(name_korean="김철수", generation=DiasporaGeneration.FIRST)
    ]
    items.append(photo)
    
    # 편지
    letter = MetadataFactory.create_for_document(
        title="어머니께 보내는 편지",
        file_path=f"{input_dir}/letter.jpg",
        width=2480, height=3508,
        transcription="사랑하는 어머니께...",
        material_type=MaterialType.LETTER
    )
    letter.dublin_core.date = "1975"
    letter.tags = ["편지", "LA", "향수"]
    items.append(letter)
    
    print(f"생성: {len(items)}개")
    
    
    # ===== 3. 관계 추론 =====
    print("\n3. 관계 추론")
    
    relations = engine.infer_relations(items)
    print(f"관계: {len(relations)}개")
    
    
    # ===== 4. 지식그래프 구축 =====
    print("\n4. 지식그래프 구축")
    
    for item in items:
        kg.build_from_metadata(item)
    kg.build_from_relations(relations)
    
    stats = kg.get_statistics()
    print(f"노드: {stats['nodes']}, 엣지: {stats['edges']}")
    
    
    # ===== 5. 저장 =====
    print("\n5. 저장")
    
    for item in items:
        result = storage.save(item)
        print(f"  저장: {item.dublin_core.title[:20]}...")
    
    
    # ===== 6. 검색 테스트 =====
    print("\n6. 검색")
    
    results = storage.search("가족")
    print(f"'가족' 검색: {results['total']}건")
    
    related = kg.get_related_items(items[0].item_id)
    print(f"관련 자료: {len(related)}개")
    
    
    # ===== 7. 내보내기 =====
    print("\n7. 내보내기")
    
    engine.export_relations(f"{output_dir}/relations.json")
    kg.export_graph(f"{output_dir}/graph.json")
    
    
    # ===== 8. 정리 =====
    kg.disconnect()
    storage.disconnect_all()
    
    print("\n완료!")
    return {"items": len(items), "relations": len(relations)}


# 실행
if __name__ == "__main__":
    result = run_pipeline("data/input", "data/output")
    print(f"결과: {result}")
```
